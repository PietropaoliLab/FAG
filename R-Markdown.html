Preparing data

m <- melt(Mouthwash[c(4:6, 8,9, 11, 12)], id.vars = c("Group"))
m$Timing <- str_sub(m$variable, start = -2)

m$variable <- gsub("_T0", "", m$variable)
m$variable <- gsub("_T1", "", m$variable)
m$variable <- gsub("_T2", "", m$variable)

my_comparisons <- list( c("T0", "T1"), c("T1", "T2"), c("T0", "T2") )
m$Group <- factor(m$Group, levels = c("FLUORO", "FAG"), labels = c("SF", "FAG"))
